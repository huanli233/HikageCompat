fun <T> UpdateScope(state: State<T>, performer: Hikage.UpdateScopePerformer<T, LP>.() -> Unit)

@Hikageable
fun <T> UpdateScope(state: State<T>, diff: Hikage.UpdateScopeDiffContext<T>.() -> Boolean, performer: Hikage.UpdateScopePerformer<T, LP>.() -> Unit)

Observes the given state and re-runs the performer block whenever the state updates, conditioned by the diff block. Each time the state updates, if the diff block returns true (meaning no update needed), the previously added views generated by this scope will be retained. If diff returns false, previous views are removed, and new views are added based on the updated state.
